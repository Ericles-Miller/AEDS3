utiliza um array auxiliar para contar quantas vezes cada valor ocorre no array a ser ordenado. 
Neste caso, o valor a ser ordenado é tratado como índice do array auxiliar.

#define max 100

void counting_sort(int *v, int n)
{
    int i,j,k;
    int baldes[max];

    for(i=0; i<max; i++)
    {
        baldes[v[i]]++;
    }

    for(i=0, j=0; j<max; j++)
    {
        for(k=baldes[j]; k>0; k--)
        v[i++]= j;
    }
}
